'If the user doesn't specify whether or not they want to the monoisotopic convention or not,
'then use:
Const conventionForUseMostCommonIspotopesDefault As Boolean = False

'This is VBA code for a molecular weight add-in for Microsoft Excel.
'For more information, see thread "Creating Excel Add-ins: Molecular Weight Example (Update #1)"
'on the Science Philosophy Chat Forums:
'   http://www.sciencechatforum.com/viewtopic.php?f=15&t=26204
'The original author hereby releases this code to the public domain.
'Ideally copies and redistributions of this code will include these notices and the link back
'to the Science Philosophy Chat Forum thread.
'
'Data used in this add-in comes from NIST at:
'   http://physics.nist.gov/cgi-bin/Compositions/stand_alone.pl?ele=&ascii=ascii2&isotype=all
'
'Custom entries can be added into this code.  See the functions
'   MWICustom   (for entire names)
'and
'   AMUCustom   (for custom atom names, function group names, isotope names, charge names, etc.)
'for details.


Function MW(chemicalSpecies As String, Optional useMostCommonIsotopes As Boolean = conventionForUseMostCommonIspotopesDefault) As Double
    MW = MWI(chemicalSpecies, useMostCommonIsotopes)
End Function
Function MWICustom(chemicalSpecies As String, Optional useMostCommonIsotopes As Boolean = conventionForUseMostCommonIspotopesDefault) As Double
    'Allows you to specify custom molecular names.
    'If you instead wish to specify a custom atomic unit,
    'add it to AMUCustom instead.
    'Units in this name will be recognized ONLY if the user calls it specifically on them,
    '   Example:    "water" is defined below, so if the user types =MW("water"), they'll get this definition.
    '               But if they type =MW("water2"), they will NOT get twice water's weight.
    '   Example:    "T" is defined in AMUCustom, so if the user types =MW("T"), they'll get that definition.
    '               And if they type =MW("T"), they WILL get twice tritium's atomic weight.
    'In general, define things here if they are a molecule's name, but define them in AMUCustom
    'if they are names of isotopes, atoms, or functional groups, e.g. CH3 is sometimes abbreviated as "Me" (since it's a methyl group),
    'so "Me" should be defined in AMUCustom.
    
    MWICustom = ErrorValue
    
    Dim lowerCaseChemicalSpecies As String
    lowerCaseChemicalSpecies = LCase(chemicalSpecies)
    
    Select Case lowerCaseChemicalSpecies    'put names that aren't case sensitive here
                                            '   Example:    "Water", "water", "WATER", and "WaTeR"
                                            '               all reduce to "water" here
        Case "water"
            MWICustom = MWI("H2O", useMostCommonIsotopes)
            'Water is H2O, so if user tries to call for the molecular weight of water,
            'we'll redirect them to the molecular weight of H2O.
            'We redirect with their provided argument for useMostCommonIsoptopes so that
            'they either get prevalence-weighted or most common isotopes as they specify.
    End Select
    
    If MWICustom <> ErrorValue Then
        Exit Function
    End If
    
    Select Case chemicalSpecies  'put names that are case sensitive here
        Case "water"
            MWICustom = MWI("H2O", useMostCommonIsotopes)
        Case "MEA"
            MWICustom = MWI("C2H7NO", useMostCommonIsotopes)
        Case "MEAH+"
            MWICustom = MWI("C2H8NO+", useMostCommonIsotopes)
        Case "MEACOO-"
            MWICustom = MWI("C3H6NO3-", useMostCommonIsotopes)
    End Select
End Function
Function MWI(chemicalSpecies As String, Optional useMostCommonIsotopes As Boolean = conventionForUseMostCommonIspotopesDefault) As Double
    MWI = MWICustom(chemicalSpecies, useMostCommonIsotopes)
    If MWI <> ErrorValue Then
        Exit Function
    End If
    
    Dim chem As String
    chem = SanitizeChemicalFormula(chemicalSpecies)
    
    MWI = MWICustom(chem, useMostCommonIsotopes)
    If MWI <> ErrorValue Then
        Exit Function
    End If

    If Len(chem) = 0 Then
        GoTo ErrorAssign
    End If
    
    Dim tempChem As String
    tempMW = 0
    tempAMU = 0
    i = 1
    Do While i <= Len(chem)
        If IsCap(Mid(chem, i, 1)) Or Mid(chem, i, 1) = "!" Then
            tempChem = ""
            If Mid(chem, i, 1) = "!" Then
                i = i + 1
                Do While i < Len(chem) And IsNumeric(Mid(chem, i, 1))
                    tempChem = tempChem & Mid(chem, i, 1)
                    i = i + 1
                Loop
            End If
            tempChem = tempChem & Mid(chem, i, 1)
            Do While i < Len(chem)
                If IsLow(Mid(chem, i + 1, 1)) Then
                    i = i + 1
                    tempChem = tempChem & Mid(chem, i, 1)
                Else
                    Exit Do
                End If
            Loop
            tempAMU = AMU(tempChem, useMostCommonIsotopes)
            If tempAMU = ErrorValue Then
                GoTo ErrorAssign
            End If
            coefficient1 = ""
            coefficient2 = ""
            sep = ""
            Do While i < Len(chem)
                If IsNum(Mid(chem, i + 1, 1)) Then
                    i = i + 1
                    coefficient1 = coefficient1 & Mid(chem, i, 1)
                ElseIf Mid(chem, i + 1, 1) = "." Then
                    sep = "."
                    i = i + 1
                    Exit Do
                ElseIf Mid(chem, i + 1, 1) = "/" Then
                    sep = "/"
                    i = i + 1
                    Exit Do
                Else
                    Exit Do
                End If
            Loop
            Do While i < Len(chem)
                If IsNum(Mid(chem, i + 1, 1)) Then
                    i = i + 1
                    coefficient2 = coefficient2 & Mid(chem, i, 1)
                Else
                    Exit Do
                End If
            Loop
            Select Case sep
                Case ""
                    coefficient = coefficient1
                Case "."
                    coefficient = coefficient1 + (coefficient2 * (10 ^ (-Len(coefficient2))))
                Case "/"
                    If Len(coefficient2) > 0 And coefficient2 > 0 Then
                        coefficient = coefficient1 / coefficient2
                    Else
                        GoTo ErrorAssign
                    End If
            End Select
            If coefficient = "" Then
                coefficient = 1
            End If
            tempMW = tempMW + tempAMU * coefficient
        ElseIf Mid(chem, i, 1) = "(" Then
            paraCount = 1
            j = 1
            Do While i + j <= Len(chem)
                Select Case Mid(chem, i + j, 1)
                    Case "("
                        paraCount = paraCount + 1
                    Case ")"
                        paraCount = paraCount - 1
                    Case Else
                        'nothing
                End Select
                If paraCount = 0 Then
                    Exit Do
                End If
                j = j + 1
            Loop
            If paraCount <> 0 Then
                GoTo ErrorAssign
            End If
            tempRecur = RecursiveMW(Mid(chem, i + 1, j - 1), useMostCommonIsotopes)
            If tempRecur = ErrorValue Then
                GoTo ErrorAssign
            End If
            i = i + j
            
            coefficient1 = ""
            coefficient2 = ""
            sep = ""
            Do While i < Len(chem)
                If IsNum(Mid(chem, i + 1, 1)) Then
                    i = i + 1
                    coefficient1 = coefficient1 & Mid(chem, i, 1)
                ElseIf Mid(chem, i + 1, 1) = "." Then
                    sep = "."
                    i = i + 1
                    Exit Do
                ElseIf Mid(chem, i + 1, 1) = "/" Then
                    sep = "/"
                    i = i + 1
                    Exit Do
                Else
                    Exit Do
                End If
            Loop
            Do While i < Len(chem)
                If IsNum(Mid(chem, i + 1, 1)) Then
                    i = i + 1
                    coefficient2 = coefficient2 & Mid(chem, i, 1)
                Else
                    Exit Do
                End If
            Loop
            Select Case sep
                Case ""
                    coefficient = coefficient1
                Case "."
                    coefficient = coefficient1 + (coefficient2 * (10 ^ (-Len(coefficient2))))
                Case "/"
                    If Len(coefficient2) > 0 And coefficient2 > 0 Then
                        coefficient = coefficient1 / coefficient2
                    Else
                        GoTo ErrorAssign
                    End If
            End Select
            If coefficient = "" Then
                coefficient = 1
            End If

            tempMW = tempMW + tempRecur * coefficient
        Else
            GoTo ErrorAssign
        End If
        i = i + 1
    Loop
    MWI = tempMW
    Exit Function
ErrorAssign:
    MWI = ErrorValue()
End Function

Private Function IsCap(letter As String) As Boolean
    '65 (A) to 90 (Z), 43 (+), 45 (-)
    ASCNumber = Asc(letter)
    If (ASCNumber > 64 And ASCNumber < 91) Or ASCNumber = 43 Or ASCNumber = 45 Then
        IsCap = True
    Else
        IsCap = False
    End If
End Function
Private Function IsLow(letter As String) As Boolean
    '97 (a) to 122 (z)
    If Asc(letter) > 96 And Asc(letter) < 123 Then
        IsLow = True
    Else
        IsLow = False
    End If
End Function
Private Function IsNum(letter As String) As Boolean
    '48 (0) to 57 (9)
    If Asc(letter) > 47 And Asc(letter) < 58 Then
        IsNum = True
    Else
        IsNum = False
    End If
End Function
Private Function SanitizeChemicalFormula(formula As String)  'make private later
    Dim i As Integer
    Dim tempForm As String
    tempForm = formula
    i = 1
    Do While i <= Len(tempForm)
        Select Case Mid(tempForm, i, 1)
            Case " "  'remove
                If i = 1 Then
                    tempForm = Right(tempForm, Len(tempForm) - 1)
                ElseIf i = Len(formula) Then
                    tempForm = Left(tempForm, Len(tempForm) - 1)
                Else
                    tempForm = Left(tempForm, i - 1) & Right(tempForm, Len(tempForm) - i)
                End If
            Case "_"  'remove
                If i = 1 Then
                    tempForm = Right(tempForm, Len(tempForm) - 1)
                ElseIf i = Len(formula) Then
                    tempForm = Left(tempForm, Len(tempForm) - 1)
                Else
                    tempForm = Left(tempForm, i - 1) & Right(tempForm, Len(tempForm) - i)
                End If
            Case "\"  'remove
                If i = 1 Then
                    tempForm = Right(tempForm, Len(tempForm) - 1)
                ElseIf i = Len(formula) Then
                    tempForm = Left(tempForm, Len(tempForm) - 1)
                Else
                    tempForm = Left(tempForm, i - 1) & Right(tempForm, Len(tempForm) - i)
                End If
            Case "'"  'remove
                If i = 1 Then
                    tempForm = Right(tempForm, Len(tempForm) - 1)
                ElseIf i = Len(formula) Then
                    tempForm = Left(tempForm, Len(tempForm) - 1)
                Else
                    tempForm = Left(tempForm, i - 1) & Right(tempForm, Len(tempForm) - i)
                End If
            Case "^"  'remove
                If i = 1 Then
                    tempForm = Right(tempForm, Len(tempForm) - 1)
                ElseIf i = Len(formula) Then
                    tempForm = Left(tempForm, Len(tempForm) - 1)
                Else
                    tempForm = Left(tempForm, i - 1) & Right(tempForm, Len(tempForm) - i)
                End If
            Case "["
                If i = 1 Then
                    tempForm = "(" & Right(tempForm, Len(tempForm) - 1)
                ElseIf i = Len(formula) Then
                    tempForm = Left(tempForm, Len(tempForm) - 1) & "("
                Else
                    tempForm = Left(tempForm, i - 1) & "(" & Right(tempForm, Len(tempForm) - i)
                End If
                i = i + 1
            Case "<"
                If i = 1 Then
                    tempForm = "(" & Right(tempForm, Len(tempForm) - 1)
                ElseIf i = Len(formula) Then
                    tempForm = Left(tempForm, Len(tempForm) - 1) & "("
                Else
                    tempForm = Left(tempForm, i - 1) & "(" & Right(tempForm, Len(tempForm) - i)
                End If
                i = i + 1
            Case "{"
                If i = 1 Then
                    tempForm = "(" & Right(tempForm, Len(tempForm) - 1)
                ElseIf i = Len(formula) Then
                    tempForm = Left(tempForm, Len(tempForm) - 1) & "("
                Else
                    tempForm = Left(tempForm, i - 1) & "(" & Right(tempForm, Len(tempForm) - i)
                End If
                i = i + 1
            Case "]"
                If i = 1 Then
                    tempForm = ")" & Right(tempForm, Len(tempForm) - 1)
                ElseIf i = Len(formula) Then
                    tempForm = Left(tempForm, Len(tempForm) - 1) & ")"
                Else
                    tempForm = Left(tempForm, i - 1) & ")" & Right(tempForm, Len(tempForm) - i)
                End If
                i = i + 1
            Case ">"
                If i = 1 Then
                    tempForm = ")" & Right(tempForm, Len(tempForm) - 1)
                ElseIf i = Len(formula) Then
                    tempForm = Left(tempForm, Len(tempForm) - 1) & ")"
                Else
                    tempForm = Left(tempForm, i - 1) & ")" & Right(tempForm, Len(tempForm) - i)
                End If
                i = i + 1
            Case "}"
                If i = 1 Then
                    tempForm = ")" & Right(tempForm, Len(tempForm) - 1)
                ElseIf i = Len(formula) Then
                    tempForm = Left(tempForm, Len(tempForm) - 1) & ")"
                Else
                    tempForm = Left(tempForm, i - 1) & ")" & Right(tempForm, Len(tempForm) - i)
                End If
                i = i + 1
            Case Else
                i = i + 1
        End Select
    Loop
    
    SanitizeChemicalFormula = tempForm
End Function

Private Function AMU(element As String, useMostCommonIsotopes As Boolean) As Double
    'The AMU function had to be split up because it was too long for VBA
    'Isotope atomic weights from NIST at
    'http://physics.nist.gov/cgi-bin/Compositions/stand_alone.pl?ele=&ascii=ascii2&isotype=all
    'Data retrieved and parsed by an automated system: 2014-3-7 at 19:43:43 (UTC)

    AMU = AMUCustom(element, useMostCommonIsotopes)
    If AMU <> ErrorValue Then
        Exit Function
    End If

    Dim fullElement As String

    If IsNum(Left(element, 1)) Or Not useMostCommonIsotopes Then
        fullElement = element
    Else
        fullElement = MostCommonIsotope(element) & element
    End If

    If AMU <> ErrorValue Then
        Exit Function
    End If
    AMU = AMU1(fullElement)

    End If

Private Function AMUCustom(element As String, useMostCommonIsotopes As Boolean) As Double
    'Add in custom entities here.
    'Note that they must conform to the same rules that normal element symbols must, i.e.:
    '    -start with a capital letter
    '    -contain nothing but lower case letters after that

    AMUCustom = ErrorValue()
    Select Case element
        Case "-"
            AMUCustom = 0.00054857990946  'Negative charges basically have an atomic weight equal to an eletron's mass
        Case "+"
            AMUCustom = -0.00054857990946  'Positive charges basically have an atomic weight equal to the negative of an eletron's mass
        Case "D"
            AMUCustom = AMU("2H", useMostCommonIsotopes)  'Deuterium, the second isotope of Hydrogen
        Case "T"
            AMUCustom = AMU("3H", useMostCommonIsotopes)  'Tritium, the third isotope of Hydrogen
        Case "Me"
            AMUCustom = MWI("CH3", useMostCommonIsotopes)  'Methyl groups, CH3, are sometimes "Me"
    End Select
End Function
